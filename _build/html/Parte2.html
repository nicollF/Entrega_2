def detect_outliers_zscore(data):
    outliers = []
    thres = 3
    mean = np.mean(data)
    std = np.std(data)
    for idx, value in enumerate(data):
        z_score = (value - mean) / std
        if np.abs(z_score) > thres:
            outliers.append(idx)
    
    return outliers

out=detect_outliers_zscore(gasolina["valor"])
